    async function runAnalysis(){
    if(!API_KEY){
    alert('Please enter and save your Anthropic API key first (shown above the form).');
    document.getElementById('api-key-input').focus();
    return;
    }
    var btn=document.getElementById('analyze-btn');
    btn.disabled=true;
    document.getElementById('btn-text').textContent='Analysing...';
    document.getElementById('btn-spinner').style.display='inline-block';
    document.getElementById('ai-result-wrap').style.display='block';
    var out=document.getElementById('ai-out');
    out.innerHTML='<span style="color:var(--text3);font-style:italic">The AI physiologist is reviewing your metrics...</span>';
    var name=document.getElementById('p-name').value||'Player';
    var pos=document.getElementById('p-pos').value;
    var sleep=document.getElementById('sl-sleep').value;
    var rpe=document.getElementById('sl-rpe').value;
    var sore=document.getElementById('sl-sore').value;
    var mental=document.getElementById('sl-mental').value;
    var days=document.getElementById('sl-days').value;
    var pain=document.getElementById('p-pain').value||'none';
    var prompt="You are a certified sports physiologist with 15 years of experience in football injury prevention for grassroots and semi-professional teams. You specialise in the acute:chronic workload ratio (ACWR) method and position-specific injury risk profiling.\n\nAnalyse this player's daily wellness log:\n\nPlayer: "+name+"\nPosition: "+pos+"\nSleep last night: "+sleep+" hours\nYesterday's RPE: "+rpe+"/10\nMuscle soreness: "+sore+"/10\nMental fatigue: "+mental+"/10\nHigh-load training days this week: "+days+" days\nReported pain/discomfort: "+pain+"\n\nPosition-specific injury mapping to apply:\n- Striker/Winger: hamstring, groin (sprint-dominant)\n- Central Midfielder: hip flexor, knee (high volume running)\n- Defender: knee, ankle (contact + aerial)\n- Goalkeeper: shoulder, lower back\n\nDetect cumulative fatigue: if RPE 7+ for 3+ days, flag as high-risk even if today's individual metrics look moderate. Sleep below 6 hours = +30% injury risk multiplier. Mental fatigue above 7 = CNS fatigue indicator.\n\nRespond with exactly:\n1. First line: 'Readiness: X/10' (and nothing else on that line)\n2. 'Injury risk: Low/Medium/High'\n3. Specific injury risk — body part and reason (1 sentence)\n4. Today's training recommendation (e.g. 50% load, rest, full intensity)\n5. One specific drill modification or recovery protocol\n\nBe concise (under 130 words). Sound like a professional, not a chatbot. Be specific, not vague.";
    try{
    var resp=await fetch('https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-flash:generateContent?key='+API_KEY,{
    method:'POST',
    headers:{'Content-Type':'application/json'},
    body:JSON.stringify({contents:[{parts:[{text:prompt}]}]})
    });
    var data=await resp.json();
    if(data.error){throw new Error(data.error.message);}
    var text=data.candidates[0].content.parts[0].text;
    out.textContent=text;
    var scoreMatch=text.match(/Readiness:\s*(\d+(?:\.\d+)?)/i);
    if(scoreMatch){document.getElementById('score-num').textContent=parseFloat(scoreMatch[1]).toFixed(1);}
    var riskMatch=text.match(/Injury risk:\s*(Low|Medium|High)/i);
    var riskLevel=riskMatch?riskMatch[1]:'Medium';
    var pill=document.getElementById('risk-pill');
    pill.textContent=riskLevel+' Risk';
    pill.className='risk-pill '+(riskLevel==='High'?'rp-high':riskLevel==='Medium'?'rp-med':'rp-low');
    }catch(e){
    out.textContent='API error: '+e.message+'. Please check your Gemini API key is correct. Get your free key at aistudio.google.com';
    out.style.color='var(--red)';
    }
    btn.disabled=false;
    document.getElementById('btn-text').textContent='Analyse with AI Physio';
    document.getElementById('btn-spinner').style.display='none';
    }
